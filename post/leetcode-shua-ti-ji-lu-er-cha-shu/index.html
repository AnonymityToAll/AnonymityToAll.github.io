1.左叶子之和

code
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        if (!root){
            return 0;
        }
        queue<TreeNode*> Q;
        Q.push(root);
        int s = 0;
        while(!Q.empty()){
            TreeNode* node = Q.front();
            TreeNode* temp;
            Q.pop();
            if(node->left){
                temp = node->left;
                if((temp->left == NULL) && (temp->right == NULL)){
                    s += temp->val;
                }
                Q.push(temp);
            }
            if(node->right){
                Q.push(node->right);
            }
        }
        return s;
    }
};
nice
  1. queue的基本操作：pop,back,front,push,empty,size

3.二叉树深度

code：
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(!root){
            return 0;
        }
        int len = 0;
        queue<TreeNode*> Q;
        Q.push(root);
        while(!Q.empty()){
            int n = Q.size();
            for(int i=0; i<n; i++){
                TreeNode* node = Q.front();
                Q.pop();
                if(node->left){
                    Q.push(node->left);
                }
                if(node->right){
                    Q.push(node->right);
                }
            }
            len++;
        }
        return len;
    }
};
nice:
  1. 记录二叉树深度需要在层次遍历的基础上多一个循环

        if(!root){
            return NULL;
        }
        helper(root);
        return root;
    }
    void helper(TreeNode* root){
        if(!root){
            return;
        }
        TreeNode* node = root->left;
        root->left = root->right;
        root->right = node;
        helper(root->left);
        helper(root->right);
    }
};

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        if(!t1 && !t2){
            return NULL;
        }
        queue<TreeNode*> Q1;
        Q1.push(t1);
        queue<TreeNode*> Q2;
        Q2.push(t2);
        if(t1 && t2){
            t1->val = t2->val + t1->val;
        }
        else if(t2){
            return t2;
        }
        else if(t1){
            return t1;
        }
        while(!Q1.empty() && !Q2.empty()){
            TreeNode* node1 = Q1.front();
            TreeNode* node2 = Q2.front();
            Q1.pop();
            Q2.pop();
            if(node1->left && node2->left){
                node1->left->val = node1->left->val + node2->left->val;
                Q1.push(node1->left);
                Q2.push(node2->left);
            }
            else if(node2->left){
                node1->left = node2->left;
            }
            if(node1->right && node2->right){
                node1->right->val = node1->right->val + node2->right->val;
                Q1.push(node1->right);
                Q2.push(node2->right);
            }
            else if(node2->right){
                node1->right = node2->right;
            }
        }
        return t1;
    }
};
nice:
  1. 用递归之前先想想递归行不行

7.二叉搜索树的范围和

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int rangeSumBST(TreeNode* root, int L, int R) {
        if(!root){
            return 0;
        }
        int s = 0;
        int t = 0;
        vector<int> trace;
        helper(root, trace);
        for (int i=0; i<trace.size(); i++){
            if(trace[i]==L){
                t = 1;
            }
            s += t*trace[i];
            if(trace[i]==R){
                break;
            }
        }
        return s;
    }
    void helper(TreeNode* root, vector<int>& trace){
        if(!root){
            return;
        }
        helper(root->left, trace);
        trace.push_back(root->val);
        helper(root->right, trace);
    }
};

nice:

一种思路有问题就换一种思路

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevel = 0;

    int maxDepth(TreeNode* root) {
        if (!root) {
            return 0;
        }
        dfs(root, 1);
        return maxLevel;
    }

    void dfs(TreeNode* root, int level) {
        if (!root)
            return;
        if (level > maxLevel) maxLevel = level;
        dfs(root->left, level + 1);
        dfs(root->right, level + 1);
    }
};
nice:
  1. pull是没有left和right的，会报错
  2. 宽搜
  3. 把层数放在函数里面表示

    }
};

    Node() {}

    Node(int _val) {
        val = _val;
    }

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};

class Solution {
public:
    vector<int> postorder(Node* root) {
        vector<int> result;
        if(!root){
            result;
        }
        helper(root, result);
        return result;
    }
    void helper(Node* root, vector<int>& result){
        if(!root){
            return;
        }
        vector<Node*> chi = root->children;
        for(int i=0; i<chi.size(); i++){
            helper(chi[i], result);
        }
        result.push_back(root->val);
    }
};

10. 二叉搜索树中查找元素

code:
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        if(!root){
            return NULL;
        }
        TreeNode* result = NULL;
        helper(root, val, result);
        return result;
    }
    void helper(TreeNode* root, int val, TreeNode*& result){
        if(!root){
            return;
        }
        int value = root->val;
        if(val == value){
            result = root;
        }
        else if(val < value){
            helper(root->left, val, result);
        }
        else if(val > value){
            helper(root->right, val, result);
        }
    }
};
nice:
  1. 一定要注意最后要求的返回值是不是形参
  2. 递归就不要返回值了

11. 二叉搜索树查找第k大的数

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* result = NULL;
    int kthLargest(TreeNode* root, int k) {
        if(!root){
            return 0;
        }
        vector<TreeNode*> trace;
        helper(root, k, trace);
        return result->val;
    }
    void helper(TreeNode* root, int k,
    vector<TreeNode*>& trace){
        if(trace.size() == k){
            result = trace.back();
            return;
        }
        if(!root){
            return;
        }
        helper(root->right, k, trace);
        trace.push_back(root);
        helper(root->left, k, trace);
    }
};
nice:
  1. 递归中的return只是其中一个小的过程的return

12.最近公共祖先

code:
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(!root || !p || !q){
            return NULL;
        }
        TreeNode* result = root;
        // 找到从根节点到p的路径
        vector<TreeNode*> vec;
        vector<TreeNode*> tracep;
        DFS(root, p, tracep, vec);
        // 找到从根节点到q的路径
        vec.clear();
        vector<TreeNode*> traceq;
        DFS(root, q, traceq, vec);
        // 正着遍历两个vector，到最后一个相等的
        int len_p = tracep.size();
        int len_q = traceq.size();
        if(len_p > len_q){
            for(int i=0; i<len_q; i++){
                if(tracep[i]->val == traceq[i]->val){
                    result = tracep[i];
                }
            }
        }
        else{
            for(int i=0; i<len_p; i++){
                if(tracep[i]->val == traceq[i]->val){
                    result = tracep[i];
                }
            }
        }
        return result;
    }
    void DFS(TreeNode* root, TreeNode* node, vector<TreeNode*>& trace
    , vector<TreeNode*>& vec){
        if(!root){
            return;
        }
        vec.push_back(root);
        if(root == node){
            //vec.push_back(node);
            trace = vec;
            return;
        }
        DFS(root->left, node, trace, vec);
        DFS(root->right, node, trace, vec);
        vec.pop_back();
    }
};
nice:
  1. 变量命名一定要明确，要不然很容易误用
  2. pop_back()要放在最后面
