最长上升子序列

错误示例：

#include<iostream>
#include<stdio.h>
#include<vector>
 
using namespace std; //动态规划
 
int main(){
    int N;
    int result = 0;
    while(cin >> N){
        vector<int> dp(N, 1);
        vector<int> nums(N, 0);
        for (int i=0; i<N; i++){
            cin >> nums[i];
            for (int j=0; j<i; j++){
                if(nums[j] < nums[i]){
                    dp[i] = max(dp[i], dp[j]+1);
                }
            } 
            result = max(result, dp[i]);
        }
        cout<<result<<endl;
    }
    
    return 0;
}

正确示例2：

#include<iostream>
#include<stdio.h>
#include<vector>
 
using namespace std; //动态规划
 
int main(){
    int N;
    while(cin >> N){
        vector<int> array(N+1, 0);
        vector<int> nums(N, 0);
        int len = 1;
        cin >> nums[0];
        array[len] = nums[0];
        for (int i=1; i<N; i++){
            cin >> nums[i];
            if(nums[i]>array[len]){
                array[++len] = nums[i];
            }
            else{
                int left = 1;
                int right = len;
                int pos = 0;
                while(left <= right){
                    int mid = (left + right) / 2;
                    if (array[mid] < nums[i]){
                        pos = mid;
                        left = mid + 1;
                    }
                    else{
                        right = mid - 1;
                    }
                }
                array[pos+1] = nums[i];
            }
        }
        cout<<len<<endl;
    }
    
    return 0;
}

统计字符串字符出现次数

#include<stdio.h>
#include<string>
#include<vector>
#include<iostream>
#include<algorithm>

using namespace std;

int main(){
    string str;
    string result = "";
    while(cin >> str){ 
        //cout << "*****" << str << "*****" << endl;
        int max = 0;
        int a[256] = {0};
        for(int i=0; str[i]!='\0'; i++){
            int num = str[i];
            a[num]++;
            //cout << str[i] << "*";
        }
        for (int j=0; j<256; j++){
            if (max < a[j]){
                max = a[j];
            }
        }
        while(max!=0){
            for (int k=0; k<256; k++){
                if (max == a[k]){
                    char c = k;
                    result.append(1, c);
                    //cout << result << "**";
                }
            }
            max--;
        }
        cout << result << endl;
        result = "";
        str = "";
    }
    
    return 0;
}

注意：

  1. 使用while(cin >> str){}。
  2. 因为有多次输入，所以全局变量要在下次使用之前清零。
