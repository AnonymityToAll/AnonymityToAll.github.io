2.生成n对合法括号

code：
class Solution {
public:
    vector<string> generateParenthesis(int n) {
        vector<string> result;
        if (n == 0){
            result.push_back("");
            return result;
        }
        helper("", result, n, n);
        return result;
    }
    void helper(string item, vector<string>& result, int left, int right){
        if(0 == right && left == 0){
            result.push_back(item);
            return;
        }
        if(left > 0){
            helper(item+'(', result, left-1, right);
        }
        if(left < right){
            helper(item+')', result, left, right-1);
        }
    }
};
nice:
