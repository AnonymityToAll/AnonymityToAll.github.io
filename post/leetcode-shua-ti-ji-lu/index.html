1o21 删除最外层的括号

class Solution {
public:
	string removeOuterParentheses(string S) {
		string res = "";
		stack<char> mystack;
		for (int i = 0; i < S.size(); i++) {
			if (S[i] == ')')
				mystack.pop();
			if (!mystack.empty())
				res+=S[i];
			if (S[i] == '(')
				mystack.push('(');
		}
		return res;
	}
};

遍历括号集，如果是右括号，说明栈里面至少有一个左括号，而且这个左括号一定已经加到栈里面了，所以弹出。
如果，栈是非空的，说明原本栈里面有两个连续的括号，那么说明剩下的左括号跟现在的右括号一块一定是要输出的结果的顺序。
抓住的结果的特点是，外层括号（要删除的）一定是连续的两个一块，也就是必然是"(("或者"))"。策略是先把外括号放进来一个，再进行内括号的处理的思路

class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> mystack;
        string result = "";
        for(int i=0; i<S.length(); i++){
            if(!mystack.empty() && S[i] == mystack.top()){
                mystack.pop();
                continue;
            }
            mystack.push(S[i]);
        }
        while(!mystack.empty()){
            result += mystack.top();
            mystack.pop();
        }
        reverse(result.begin(), result.end());
        return result;
    }
};

遇到的问题：超出边界了，要先判断mystack是非空的，同时要放在&&符号前面的位置才可以

class Solution {
public:
    int calPoints(vector<string>& ops) {
        stack<int> mystack;
        int result = 0;
        int num = 0;
        for(int i=0; i<ops.size(); i++){
            if(ops[i][0] == '-'){
                num = stringToInt2(ops[i]);
                mystack.push(num);
                //printf("**%d\n",num);
            }
            else if(ops[i] == "C"){
                mystack.pop();
            }
            else if(ops[i] == "D"){
                num = mystack.top() * 2;
                mystack.push(num);
            }
            else if(ops[i] == "+"){
                int temp = mystack.top();
                mystack.pop();
                num = mystack.top();
                mystack.push(temp);
                mystack.push(temp + num);
            }
            else{
                num = stringToInt(ops[i]);
                mystack.push(num);
                //printf("***%d\n",num);
            }
        }
        while(!mystack.empty()){
            result += mystack.top();
            //printf("**%d\n",mystack.top());
            mystack.pop();
        }
        return result;
    }
private:
    int stringToInt(string S){
        int result = 0;
        for(int i=0; i<S.length(); i++){
            result = result*10 + S[i] - '0';
        }
        return result;
    }
    int stringToInt2(string S){
        int result = 0;
        for(int i=1; i<S.length(); i++){
            result = result*10 + S[i] - '0';
        }
        result = 0 - result;
        return result;
    }
};
